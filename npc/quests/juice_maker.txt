payon_in03,188,146,5	script	马雷克汉森#juice	86,{
	if (checkweight(1201,1) == 0) {
		mes "- 稍等！-";
		mes "- 目前你现在带的东西太多了 -";
		mes "- 可以把它们放进仓库 -";
		mes "- 然后再回来找我哦~. -";
		close;
	}
	if (MISC_QUEST&1) {
		mes "[商人 马雷克汉森]";
		mes "哟.";
		mes "你想找本大爷制作果汁？";
		next;
		switch(select("我要制作果汁！:了解制作果汁的材料:取消")) {
		case 1:
			mes "[商人 马雷克汉森]";
			mes "你想制作什么类型的果汁？";
			next;
			switch(select("苹果汁:香蕉汁:红萝卜汁:葡萄汁:取消")) {
			case 1:
				set .@fruit,512;
				set .@juice,531;
				break;
			case 2:
				set .@fruit,513;
				set .@juice,532;
				break;
			case 3:
				set .@fruit,515;
				set .@juice,534;
				break;
			case 4:
				set .@fruit,514;
				set .@juice,533;
				break;
			case 5:
				mes "[商人 马雷克汉森]";
				mes "嗯，有空常来哈~";
				close;
			}
			if (countitem(.@fruit) == 0 || countitem(713) == 0 || Zeny < 3) {
				mes "[商人 马雷克汉森]";
				mes "噢，额滴神啊...";
				mes "小子，你的材料不够呀。";
				mes "想制作 "+getitemname(.@juice)+" 的话，";
				mes "我需要 1 个 "+getitemname(.@fruit)+" 和 1 个空瓶。";
				mes "此外还要 3 Zeny 的手续费。";
				next;
				mes "[商人 马雷克汉森]";
				mes "等你准备好材料再来找我吧， 我给你制作美味的果♂汁。";
				close;
			}

			mes "[商人 马雷克汉森]";
			mes "你要制作多少啊？";
			next;
			switch(select("我的材料能制作多少个，就要多少个:我想自己决定制作的数量:取消")) {
			case 1:
				set .@make,countitem(.@fruit);
				if (countitem(713) < .@make) set .@make, countitem(713);
				if (Zeny/3 < .@make) set .@make, Zeny/3;
				break;
			case 2:
				mes "[商人 马雷克汉森]";
				mes "选择的数字要小于 100 如果你想取消，";
				mes "请输入 '0' 即可。";
				mes "你的材料最多可以制作 " + countitem(.@fruit) + " 瓶果汁。";
				next;
				while(1) {
					input .@input;
					if (.@input == 0) {
						mes "[商人 马雷克汉森]";
						mes "嗯，有空再来...";
						close;
					}
					else if (.@input > 100) {
						mes "[商人 马雷克汉森]";
						mes "要我一次制作100瓶不可能的啦，最多99个一次！";
						next;
					}
					else {
						break;
					}
				}
				set .@make,.@input;
				break;
			case 3:
				mes "[商人 马雷克汉森]";
				mes "嗯，有空再来...";
				close;
			}

			set .@total_zeny,3 * .@make;

			if (countitem(.@fruit) < .@make || countitem(713) < .@make || Zeny < .@total_zeny) {
				mes "[商人 马雷克汉森]";
				mes "小子，你的材料不够呀，等你准备好材料再来找我吧。";
				close;
			}
			delitem .@fruit,.@make;
			delitem 713,.@make;
			set Zeny, Zeny-.@total_zeny;
			getitem .@juice,.@make;

			mes "[商人 马雷克汉森]";
			mes "喏，拿去吧！清爽可口的果汁！";
			next;
			mes "[商人 马雷克汉森]";
			mes "嗯，有空再来...";
			close;
		case 2:
			mes "[商人 马雷克汉森]";
			mes "人类之所以有今天的发展，是因为我们获得了大自然的祝福――果实.";
			next;
			mes "[商人 马雷克汉森]";
			mes "如今生命如此繁荣，年轻的一代却不怎么吃水果，于是本大爷就想怎么让水果更受欢迎。";
			next;
			mes "[商人 马雷克汉森]";
			mes "我意识到用水果制作的果汁让年轻人更容易接受。";
			next;
			mes "# 果汁制作材料清单 #";
			mes "^CC4E5C- 苹果汁 -^000000";
			mes "苹果 x 1 个，空瓶 x 1 个，3 金币";
			mes "^E3CF57- 香蕉汁 -^000000";
			mes "香蕉 x 1 条，空瓶 x 1 个，3 金币";
			mes "^ED9121- 萝卜汁 -^000000";
			mes "红萝卜 x 1 个，空瓶 x 1 个，3 金币";
			mes "^CC00FF- 葡萄汁 -^000000";
			mes "葡萄 x 1 颗，空瓶 x 1 个，3 金币";
			close;
		case 3:
			mes "[商人 马雷克汉森]";
			mes "小子，有空帮我宣传一下啊！";
			close;
		}
	}
	else {
		mes "[商人 马雷克汉森]";
		mes "哟！";
		mes "你想找本大爷制作果汁？";
		next;
		if (select("对话:取消") == 1) {
			mes "[商人 马雷克汉森]";
			mes "人类之所以有今天的发展，是因为我们获得了大自然的祝福――果实。";
			next;
			mes "[商人 马雷克汉森]";
			mes "如今生命如此繁荣，年轻的一代却不怎么吃水果，于是本大爷就想怎么让水果更受欢迎。";
			next;
			mes "[商人 马雷克汉森]";
			mes "我意识到用水果制作的果汁让年轻人更容易接受。";
			close;
		}
		mes "[商人 马雷克汉森]";
		mes "小子，有空帮我宣传一下啊！";
		close;
	}
